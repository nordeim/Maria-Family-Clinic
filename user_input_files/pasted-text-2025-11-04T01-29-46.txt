Motion Design Rules:

Purposeful, Not Decorative: Every animation should serve a purpose (feedback, direction, relationship).
Respect Reduced Motion: Always provide prefers-reduced-motion: reduce fallbacks.
Consistent Timing: Use the same duration for similar actions across the site.
Ease Healthcare Anxiety: Avoid sudden, jarring, or aggressive animations.

Backend: tRPC Search Endpoint

TypeScript

// server/routers/search.ts
export const searchRouter = router({
  global: publicProcedure
    .input(z.object({
      query: z.string().min(2).max(100),
      limit: z.number().min(1).max(20).default(5),
    }))
    .query(async ({ input, ctx }) => {
      const { query, limit } = input;
      
      const [clinics, services, doctors] = await Promise.all([
        ctx.prisma.clinic.findMany({
          where: {
            OR: [
              { name: { contains: query, mode: 'insensitive' } },
              { district: { contains: query, mode: 'insensitive' } },
              { postal_code: { contains: query } },
            ],
          },
          take: limit,
          select: { id: true, name: true, slug: true, district: true },
        }),
        
        ctx.prisma.service.findMany({
          where: {
            OR: [
              { name: { contains: query, mode: 'insensitive' } },
              { short_description: { contains: query, mode: 'insensitive' } },
            ],
          },
          take: limit,
          select: { id: true, name: true, slug: true, category: true },
        }),
        
        ctx.prisma.doctor.findMany({
          where: {
            OR: [
              { full_name: { contains: query, mode: 'insensitive' } },
              { specialties: { hasSome: [query] } },
            ],
          },
          take: limit,
          select: { id: true, full_name: true, slug: true, specialties: true },
        }),
      ]);
      
      return { clinics, services, doctors };
    }),
});
2.3 Persistent User Context
Enhancement: Remember User Preferences & Location

TypeScript

// lib/user-context.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface UserContextState {
  lastSearchLocation: { lat: number; lng: number; label: string } | null;
  preferredClinic: string | null; // clinic slug
  recentlyViewed: {
    type: 'clinic' | 'service' | 'doctor';
    id: string;
    name: string;
    slug: string;
    viewedAt: string;
  }[];
  
  setLastSearchLocation: (location: { lat: number; lng: number; label: string }) => void;
  setPreferredClinic: (slug: string) => void;
  addRecentlyViewed: (item: Omit<RecentlyViewedItem, 'viewedAt'>) => void;
  clearRecentlyViewed: () => void;
}

export const useUserContext = create<UserContextState>()(
  persist(
    (set) => ({
      lastSearchLocation: null,
      preferredClinic: null,
      recentlyViewed: [],
      
      setLastSearchLocation: (location) => set({ lastSearchLocation: location }),
      setPreferredClinic: (slug) => set({ preferredClinic: slug }),
      
      addRecentlyViewed: (item) => set((state) => ({
        recentlyViewed: [
          { ...item, viewedAt: new Date().toISOString() },
          ...state.recentlyViewed.filter(i => i.id !== item.id), // Remove duplicates
        ].slice(0, 10), // Keep last 10
      })),
      
      clearRecentlyViewed: () => set({ recentlyViewed: [] }),
    }),
    {
      name: 'mfc-user-context',
    }
  )
);

// Form submission with optimistic update
function EnquiryForm() {
  const utils = trpc.useUtils();
  const { mutate, isLoading } = trpc.enquiries.create.useMutation({
    onMutate: async (newEnquiry) => {
      // Cancel outgoing refetches
      await utils.enquiries.list.cancel();
      
      // Snapshot current data
      const previousEnquiries = utils.enquiries.list.getData();
      
      // Optimistically update UI
      utils.enquiries.list.setData(undefined, (old) => [
        {
          id: 'temp-' + Date.now(),
          ...newEnquiry,
          status: 'pending',
          created_at: new Date().toISOString(),
        },
        ...(old || []),
      ]);
      
      // Show instant success feedback
      toast.success('Enquiry submitted! We\'ll respond within 24 hours.');
      
      return { previousEnquiries };
    },
    
    onError: (err, newEnquiry, context) => {
      // Rollback on error
      utils.enquiries.list.setData(undefined, context?.previousEnquiries);
      toast.error('Failed to submit. Please try again.');
    },
    
    onSettled: () => {
      // Refetch after mutation
      utils.enquiries.list.invalidate();
    },
  });
  
  // Form implementation
}

// hooks/use-prefers-reduced-motion.ts
export function usePrefersReducedMotion() {
  const [prefersReduced, setPrefersReduced] = useState(false);
  
  useEffect(() => {
    const mediaQuery = window.matchMedia('(prefers-reduced-motion: reduce)');
    setPrefersReduced(mediaQuery.matches);
    
    const handleChange = (e: MediaQueryListEvent) => {
      setPrefersReduced(e.matches);
    };
    
    mediaQuery.addEventListener('change', handleChange);
    return () => mediaQuery.removeEventListener('change', handleChange);
  }, []);
  
  return prefersReduced;
}

// hooks/use-keyboard-shortcuts.ts
export function useKeyboardShortcuts() {
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      // Cmd/Ctrl + K: Open search
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault();
        openSearch();
      }
      
      // Cmd/Ctrl + L: Go to locator
      if ((e.metaKey || e.ctrlKey) && e.key === 'l') {
        e.preventDefault();
        navigate('/clinics');
      }
      
      // ? : Show keyboard shortcuts help
      if (e.key === '?' && !isInputFocused()) {
        e.preventDefault();
        openShortcutsDialog();
      }
    };
    
    document.addEventListener('keydown', handleKeyDown);
    return () => document.removeEventListener('keydown', handleKeyDown);
  }, []);
}

7.1 Voice & Tone Guidelines
Missing from PRD - Critical for Healthcare

Markdown

# My Family Clinic Voice & Tone Guide

## Core Voice Attributes
- **Professional yet Approachable**: Expertise without intimidation
- **Reassuring**: Calm confidence that eases anxiety
- **Clear**: Plain language over medical jargon
- **Caring**: Empathetic and patient-focused
- **Trustworthy**: Transparent and evidence-based

## Tone Adjustments by Context

### Homepage & Marketing
Tone: Warm, welcoming, confident
- "Your trusted neighborhood healthcare partner"
- "Quality care, close to home"

### Service Descriptions
Tone: Informative, reassuring, clear
- ✅ "A routine health screening helps catch potential issues early"
- ❌ "Comprehensive diagnostic panel for pathological detection"

### Error Messages
Tone: Helpful, solution-focused, never blaming
- ✅ "We couldn't find clinics in that area. Try expanding your search radius."
- ❌ "No results found."

### Urgent/Medical Notices
Tone: Direct, clear, calm
- ✅ "For urgent medical issues, please call 995 or visit the nearest A&E"
- ❌ "Don't use this form for emergencies!"

## Language Guidelines

### Do's
- Use "you" and "your" (second person)
- Write in active voice
- Break long sentences into shorter ones
- Define medical terms when first used
- Use inclusive language (they/them when gender unknown)

### Don'ts
- Avoid medical jargon without explanation
- Don't use ALL CAPS (except acronyms like GP, A&E)
- Avoid passive voice ("The clinic can be reached" → "You can reach the clinic")
- Don't make absolute promises ("We'll cure..." → "We can help treat...")

// Geolocation with graceful fallback
function useGeolocation() {
  const [location, setLocation] = useState<GeolocationCoordinates | null>(null);
  const [error, setError] = useState<string | null>(null);
  
  const requestLocation = () => {
    if (!('geolocation' in navigator)) {
      setError('Geolocation is not supported by your browser');
      return;
    }
    
    navigator.geolocation.getCurrentPosition(
      (position) => {
        setLocation(position.coords);
        setError(null);
      },
      (error) => {
        switch (error.code) {
          case error.PERMISSION_DENIED:
            setError('Location access denied. Please search by postal code instead.');
            break;
          case error.POSITION_UNAVAILABLE:
            setError('Location information unavailable. Please search manually.');
            break;
          case error.TIMEOUT:
            setError('Location request timed out. Please try again or search manually.');
            break;
        }
      },
      {
        timeout: 10000,
        maximumAge: 300000, // Cache for 5 minutes
      }
    );
  };
  
  return { location, error, requestLocation };
}
